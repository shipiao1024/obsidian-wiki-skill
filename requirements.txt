# -----------------------------------------------------------------------------
# [微信] 微信公众号文章入库 — 必装（主要使用场景）
#   安装方式（标准）：
#     git clone --depth 1 https://github.com/bzd6661/wechat-article-for-ai.git .tools/wechat-article-for-ai
#     pip install -r .tools/wechat-article-for-ai/requirements.txt --target .python-packages
#     pip install chardet --target .python-packages
#   安装方式（中国镜像）：
#     git clone --depth 1 https://github.com/bzd6661/wechat-article-for-ai.git .tools/wechat-article-for-ai
#     pip install -r .tools/wechat-article-for-ai/requirements.txt --target .python-packages -i https://pypi.tuna.tsinghua.edu.cn/simple
#     pip install chardet --target .python-packages -i https://pypi.tuna.tsinghua.edu.cn/simple
#
#   ⚠ Camoufox 浏览器二进制（~530MB）首次运行时自动下载，下载源为 GitHub Releases。
#     中国网络可能较慢或超时，替代方案：
#     1. 手动下载：https://github.com/daijro/camoufox/releases  → 解压到 %LOCALAPPDATA%\camoufox\camoufox\Cache\
#     2. 通过代理：set HTTPS_PROXY=http://127.0.0.1:7890 后运行
#     3. 使用 ghproxy 镜像：set CAMOUFOX_MIRROR=https://ghproxy.com/ 后手动下载
# -----------------------------------------------------------------------------

# wechat-article-for-ai 的 pip 依赖（需通过其 requirements.txt 安装，此处仅作记录）：
# camoufox[geoip]     — 反检测浏览器自动化（首次运行下载 ~530MB 浏览器二进制）
# markdownify          — HTML → Markdown 转换
# beautifulsoup4       — HTML 解析
# httpx                — 异步 HTTP 客户端
# mcp                  — Model Context Protocol SDK
chardet                # 避免 Windows 下 requests 字符集后备告警

# -----------------------------------------------------------------------------
# [视频] YouTube / Bilibili / 抖音视频入库 — 按需安装
#   安装方式（标准）：pip install yt-dlp
#   安装方式（中国镜像）：pip install yt-dlp -i https://pypi.tuna.tsinghua.edu.cn/simple
# -----------------------------------------------------------------------------
yt-dlp>=2024.0         # 视频字幕/元数据提取（也可独立安装二进制）

# -----------------------------------------------------------------------------
# [视频ASR] 无字幕视频的 ASR fallback — 按需安装
#   安装方式（标准）：pip install faster-whisper
#   安装方式（中国镜像）：pip install faster-whisper -i https://pypi.tuna.tsinghua.edu.cn/simple
#
#   ⚠ 模型权重首次使用时从 Hugging Face 下载。中国镜像：
#     set HF_ENDPOINT=https://hf-mirror.com
# -----------------------------------------------------------------------------
faster-whisper>=1.0    # 无字幕视频 ASR（tiny~75MB, small~460MB, medium~1.5GB）

# -----------------------------------------------------------------------------
